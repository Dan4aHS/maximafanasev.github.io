def cc(x):
    s = ''
    while x>0:
        s = str(x%4) + s
        x //= 4
    return s

for n in range(1,100):
    n4 = cc(n)
    if len(n4) % 2 == 0:
        n4 = n4[:len(n4)//2] + '0' + n4[len(n4)//2:]
    r = int(n4)
    if r <= 180:
        print(n)
