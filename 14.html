249 Найдите десятичное число x, такое что 20 < x < 30, запись которого в системе счисления с основанием 3 заканчивается на 11

def cc(x):
    s = ''
    while x>0:
        s = str(x%3) + s
        x //= 3
    return s

for x in range(21,30):
    print(x, cc(x))
