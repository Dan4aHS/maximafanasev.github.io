Строки

...'abcde'; // разобьет строку в символы через запятую: 'a','b','c','d','e'


function func(s1, s2, s3, s4, s5) {
	return s1 + '-' + s2 + '-' + s3 + '-' + s4 + '-' + s5;
}

console.log( func(...'abcde') ); // выведет 'a-b-c-d-e'


let arr = [...'abcde']; // получим массив ['a', 'b', 'c', 'd', 'e']
