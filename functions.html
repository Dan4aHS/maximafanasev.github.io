  1. Объявление функции
    2. function showMessage() {
  alert( 'Всем привет!' );
}
  2. вызов
    3. showMessage();
  3. локальные переменные
    4. function showMessage() {
  let message = "Привет, я JavaScript!"; // локальная переменная
}
  4. доступ ко внешним
    5. let userName = 'Вася';

function showMessage() {
  let message = 'Привет, ' + userName;
  alert(message);
}

showMessage(); // Привет, Вася
  5. Функция обладает полным доступом к внешним переменным и может изменять их значение.

  6. передать внутрь функции любую информацию, используя параметры
    7. function showMessage(from, text) { // параметры: from, text
  alert(from + ': ' + text);
}
  7. значение по умолчанию
    8. function showMessage(from, text = "текст не добавлен") {
  alert( from + ": " + text );
}

showMessage("Аня"); // Аня: текст не добавлен
  8. оператор нулевого слияния
    9. function showCount(count) {
  // если count равен undefined или null, показать "неизвестно"
  alert(count ?? "неизвестно");
}
showCount(0); // 0
showCount(null); // неизвестно
showCount(); // неизвестно
  9. Возврат значения
    10. function sum(a, b) {
  return a + b;
}

let result = sum(1, 2);
alert( result ); // 3
  10. Вызовов return может быть несколько
    11. function checkAge(age) {
  if (age >= 18) {
    return true;
  } else {
    return confirm('А родители разрешили?');
  }
}
